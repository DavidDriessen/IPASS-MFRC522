100  Status ReadBlock(int block, byte *data, bool loop = true) {
101  for (int j = 0; j < keysLen; ++j) {
102  if (!prepare(loop)) {
103  return ERROR;
104  }
105 
106  if (nfc.Authenticate(0x60, block, keys[j], uid)) {
107  Status result = Read(block, data);
108  if (result != OK) {
109  return result;
110  }
111 
112  if (!nfc.Deauthenticate()) {
113  return DEAUTH;
114  }
115  return OK;
116  }
117  }
118  return AUTH;
119  }

127  Status ReadSector(int sector, byte **data, bool loop = true) {
128  sector *= 4;
129  for (int j = 0; j < keysLen; ++j) {
130  if (!prepare(loop)) {
131  return ERROR;
132  }
133  if (nfc.Authenticate(0x60, sector, keys[j], uid)) {
134  for (int block = sector; block < sector + 4; ++block) {
135  Status result = Read(block, data[block - sector]);
136  if (result != OK) {
137  return result;
138  }
139  }
140  if (!nfc.Deauthenticate()) {
141  return DEAUTH;
142  }
143  return OK;
144  }
145  }
146  return AUTH;
147  }
